56 void Camera::onEvent(Urho3D::StringHash eventType,
57  Urho3D::VariantMap& eventData)
58 {
59  if(eventType == Urho3D::E_MOUSEBUTTONDOWN
60  && eventData[Urho3D::MouseButtonDown::P_BUTTON].GetInt()
61  == Urho3D::MOUSEB_RIGHT)
62  {
63  mouseIsPressed = true;
64  GetSubsystem<Urho3D::Input>()->SetMouseVisible(false);
65  }
66  else if(eventType == Urho3D::E_MOUSEWHEEL)
67  {
68  if(eventData[Urho3D::MouseWheel::P_WHEEL].GetInt() > 0)
69  {
70  camDist /= 1.2f;
71  }
72  else if(eventData[Urho3D::MouseWheel::P_WHEEL].GetInt() < 0)
73  {
74  camDist *= 1.2f;
75  }
76  }
77  else if(eventType == Urho3D::E_MOUSEBUTTONUP
78  && eventData[Urho3D::MouseButtonUp::P_BUTTON].GetInt()
79  == Urho3D::MOUSEB_RIGHT)
80  {
81  mouseIsPressed = false;
82  GetSubsystem<Urho3D::Input>()->SetMouseVisible(true);
83  }
84  else if(eventType == Urho3D::E_MOUSEMOVE)
85  {
86  if(!mouseIsPressed || mouseEventHandled)
87  {
88  return;
89  }
90 
91  camAngleXZ -= eventData[Urho3D::MouseMove::P_DY].GetInt() / 500.f;
92  camAngleXY -= eventData[Urho3D::MouseMove::P_DX].GetInt() / 500.f;
93  if(camAngleXZ <= -1.5)
94  {
95  camAngleXZ = -1.5;
96  }
97  if(camAngleXZ >= 1.5)
98  {
99  camAngleXZ = 1.5;
100  }
101 
102  mouseEventHandled = true;
103  }
104 }
