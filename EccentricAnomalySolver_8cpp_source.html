104 double EccentricAnomalySolver::findRoot(gsl_function_fdf FDF, void* parameters,
105  double x)
106 {
107  // Settings up parameters
108  EquationParams* params = static_cast<EquationParams*>(parameters);
109 
110  // Solver Init
111  int status;
112  unsigned int currentIter(0);
113  const unsigned int maxIter(100);
114 
115  gsl_root_fdfsolver* solver(
116  gsl_root_fdfsolver_alloc(gsl_root_fdfsolver_newton));
117  gsl_root_fdfsolver_set(solver, &FDF, x);
118 
119  // Iterate
120  do
121  {
122  currentIter++;
123  gsl_root_fdfsolver_iterate(solver);
124  x = gsl_root_fdfsolver_root(solver);
125  status = gsl_root_test_residual(FDF.f(x, &params), 1e-15);
126  } while(status == GSL_CONTINUE && currentIter < maxIter);
127 
128  // Solver Freeing
129  gsl_root_fdfsolver_free(solver);
130 
131  return x;
132 }
