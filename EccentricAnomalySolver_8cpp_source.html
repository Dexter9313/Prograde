23 double EccentricAnomalySolver::solveForEllipticOrbit(double meanAnomaly,
24  double eccentricity)
25 {
26  double guess((eccentricity < 0.8 ? meanAnomaly : constant::pi));
27  boost::uintmax_t maxit = 100;
28 
29  return boost::math::tools::newton_raphson_iterate(
30  [meanAnomaly, eccentricity](const double& eccAn) {
31  return std::make_pair(eccAn - (eccentricity * sin(eccAn))
32  - meanAnomaly,
33  1 - (eccentricity * cos(eccAn)));
34  },
35  guess, 0.0, 2 * constant::pi, std::numeric_limits<double>::digits,
36  maxit);
37 }

39 double EccentricAnomalySolver::solveForParabolicOrbit(double meanAnomaly)
40 {
41  // solves D^3+3D-6M = 0
42  // Solving using Cardano's method (and the fact that there will always be
43  // one
44  // single real root)
45  // (french source, hope it's understandable anyway, better source needed,
46  // I don't find the english version is as clear in Wikipedia)
47  // https://fr.wikipedia.org/wiki/M%C3%A9thode_de_Cardan#Si_.CE.94_est_n.C3.A9gatif
48  double p(3), q(-6 * meanAnomaly);
49  double term(sqrt(q * q / 4.0 + p * p * p / 27.0));
50  return cbrt(-q / 2.0 + term) + cbrt(-q / 2.0 - term);
51 }
