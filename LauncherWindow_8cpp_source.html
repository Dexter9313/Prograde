35  engineParameters_["FullScreen"] = false;

37  engineParameters_["Sound"] = false;
38  engineParameters_["WindowResizable"] = false;
39  engineParameters_["WindowWidth"] = 800;
40  engineParameters_["WindowHeight"] = 600;
41  engineParameters_["ResourcePaths"]
42  = "data/urho3d/Data;data/urho3d/CoreData";
43 
44  // Construct a search path to find the resource prefix with two entries:
45  // The first entry is an empty path which will be substituted with
46  // program/bin directory -- this entry is for binary when it is still in
47  // build tree
48  // The second and third entries are possible relative paths from the
49  // installed program/bin directory to the asset directory -- these entries
50  // are for binary when it is in the Urho3D SDK installation location
51  if(!engineParameters_.Contains("ResourcePrefixPaths"))
52  {
53  engineParameters_["ResourcePrefixPaths"]
54  = ";../share/Resources;../share/Urho3D/Resources";
55  }
