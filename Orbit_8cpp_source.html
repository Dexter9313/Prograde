73 double Orbit::getMeanAnomalyAtUT(UniversalTime uT) const
74 {
75  // TODO(florian) : test assertion
76  // to return a sensible meanAnomaly (we don't care about returning > 2*M_PI
77  // values as big as we want mathematically but float precision Does care,
78  // so we treat potentially huge uTs like this;
79  // if uT < period, then the result will be between 0 and 2*M_PI)
80  // several loops are done for performance (we don't want one loop which
81  // loops millions of times)
82  UniversalTime uT2 = fmod(uT, period);
83  double equivUT = uT2.convert_to<double>();
84 
85  double smaCubed = parameters.semiMajorAxis * parameters.semiMajorAxis
86  * parameters.semiMajorAxis;
87  double n(sqrt(constant::G * massiveBodyMass / smaCubed));
88  return (n * equivUT) + parameters.meanAnomalyAtEpoch;
89 }

108 double Orbit::getTrueAnomalyAtUT(UniversalTime uT) const
109 {
110  if(parameters.eccentricity < 1)
111  {
112  double coeff(sqrt((1 + parameters.eccentricity)
113  / (1 - parameters.eccentricity)));
114  double eccentricAnomaly(getEccentricAnomalyAtUT(uT));
115 
116  return 2.0 * atan(coeff * tan(eccentricAnomaly / 2.0));
117  }
118 
119  if(parameters.eccentricity == 1)
120  {
121  return 2.0 * atan(getEccentricAnomalyAtUT(uT));
122  }
123 
124  double coeff(
125  sqrt((parameters.eccentricity + 1) / (parameters.eccentricity - 1)));
126  double eccentricAnomaly(getEccentricAnomalyAtUT(uT));
127 
128  return 2.0 * atan(coeff * tan(eccentricAnomaly) / 2.0);
